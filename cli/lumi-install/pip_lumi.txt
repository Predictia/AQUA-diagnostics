# pip dependencies
# Install both aqua-core and aqua-diagnostics in editable mode
# This allows modifying both repositories

# 1. First install aqua-core in editable mode
pip install -e $AQUA/[all]

# 2. Install aqua-diagnostics frontier diagnostics in editable mode
pip install -e $AQUA_DIAGNOSTICS/frontier-diagnostics/tropical_cyclones/
pip install -e $AQUA_DIAGNOSTICS/frontier-diagnostics/tropical_rainfall/

# 3. Install aqua-diagnostics with [dev] - does NOT pull aqua-core from git (assumes it's already installed)
pip install -e $AQUA_DIAGNOSTICS/[dev]
